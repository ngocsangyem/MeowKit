import { test } from '@playwright/test';
import checkoutSelectors from './common/checkoutSelectors';
import { checkoutAssertions } from './common/checkoutAssertions';
import checkoutIntercepts from './common/checkoutIntercepts';

test.describe('Checkout Flow', () => {
  test('complete checkout from product to confirmation', async ({ page }) => {
    // Step 1: Navigate to product page and add item to cart
    await page.goto('https://app.example.com/products/widget-pro');
    await checkoutSelectors.addToCartButton(page).click();

    // Step 2: Navigate to cart
    await checkoutSelectors.cartLink(page).click();
    await checkoutAssertions.shouldBeOnCart(page);

    // Step 3: Begin checkout
    await checkoutSelectors.checkoutButton(page).click();
    await checkoutAssertions.shouldBeOnCheckout(page);

    // Step 4: Fill shipping address
    await checkoutSelectors.fullNameInput(page).fill('Jane Doe');
    await checkoutSelectors.addressInput(page).fill('123 Main St');
    await checkoutSelectors.cityInput(page).fill('San Francisco');
    await checkoutSelectors.zipInput(page).fill('94102');

    // Step 5: Select payment method
    await checkoutSelectors.creditCardRadio(page).check();

    // Step 6: Confirm order
    await checkoutIntercepts.mockPlaceOrder(page);
    await checkoutSelectors.placeOrderButton(page).click();

    // Step 7: Verify order confirmation
    await checkoutAssertions.shouldShowConfirmation(page);
  });
});
